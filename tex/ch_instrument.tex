            \NC 0 \MC 233.15 \KELVIN \le \TEMP \NR
            \NC \frac{\TEMP - 233.15 \KELVIN}{40 \KELVIN}
                \MC 233.15 \KELVIN \lt \TEMP \lt 273.15 \KELVIN \NR
            \NC 1 \MC 273.15 \KELVIN \le \TEMP \NR
        \stopcases
    \stopformula
